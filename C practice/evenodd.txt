#include <stdio.h>

int main() {
    int start, end, i, count = 0;
    int sequence[100]; // Assuming a maximum of 100 numbers in the sequence

    // Get the starting and ending numbers from the user
    printf("Enter the starting number of the range: ");
    scanf("%d", &start);

    printf("Enter the ending number of the range: ");
    scanf("%d", &end);

    // Generate the sequence
    for (i = start; i <= end; i++) {
        if (i % 2 != 0) { // If the number is odd
            sequence[count++] = i;
            if (count % 5 == 0) { // After every 5 odd numbers, add an even number
                sequence[count++] = i + 1;
            }
        }
    }

    // Display the generated sequence
    printf("The generated sequence is:\n");
    for (i = 0; i < count; i++) {
        printf("%d ", sequence[i]);
    }
    printf("\n");

    return 0;
}
