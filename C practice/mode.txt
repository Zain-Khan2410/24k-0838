#include <stdio.h>
int main()
{
    int n,sum=0;
    printf("enter the number of students:");
    scanf("%d", &n);
    int marks[n];

    printf("Enter the marks of each student: \n");
    for (int i = 0; i < n; i++) {
        printf("Student %d: ", i + 1);
        scanf("%d", &marks[i]);
        
       
    }
    
    int i;

    int  max_count=0, mode=marks[0];
    for(i=0;i<n;i++){
        int count = 0;
        for(int j =0; j < n;j++ ){
           if(marks[j]==marks[i]){
            count++;
        }
        }
        if (count > max_count){
            max_count=count;
            mode=marks[i];

        }
    }
    printf("mode :%d\n",mode);
}
